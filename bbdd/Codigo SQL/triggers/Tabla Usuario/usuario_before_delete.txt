BEGIN

DECLARE varcif VARCHAR(9);
-- SET $varcif = NULL;

SET varcif = SELECT cif FROM empresa WHERE empresa.owner_username=username;

if varcif=NULL THEN
	INSERT INTO Acciones VALUE (DEFAULT,OLD.username+' ha eliminado su cuenta de usuario cliente.',NOW(),OLD.username,NULL);
END if;

if varcif<>NULL THEN
	INSERT INTO Acciones VALUE (DEFAULT,OLD.username+' ha eliminado su cuenta de usuario empresario, \n con empresa: '+varcif,
								NOW(), OLD.username, varcif);
END if;

END
